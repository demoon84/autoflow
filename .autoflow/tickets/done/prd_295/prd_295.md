## Verification

- Command: `find .autoflow/scripts runtime/board-scripts -maxdepth 1 -type f -name '*.sh' -exec bash -n {} \;`
- Command: `node -e 'const fs=require("fs"),crypto=require("crypto"); const names=["finish-ticket-owner.sh","board-guard.sh","state-db.sh","start-ticket-owner.legacy.sh","runner-common.sh","watch-board.sh","start-todo.sh"]; for (const name of names) { const a=".autoflow/scripts/"+name, b="runtime/board-scripts/"+name; if (!fs.existsSync(a)||!fs.existsSync(b)) throw new Error("missing mirror "+name); const ah=crypto.createHash("sha256").update(fs.readFileSync(a)).digest("hex"); const bh=crypto.createHash("sha256").update(fs.readFileSync(b)).digest("hex"); console.log(name, ah===bh ? "same" : "different"); }'`
